\newcommand{\AB}[2]{
    \draw[->] (-4,0)--(4,0) node[below] {$x$};
    \draw[->] (0,-1.3)--(0,1.5) node[right]{$y$};
    \draw[domain=-3.5:3.5,smooth,samples=200,thick] plot[id=#2] function { #1(x*pi)} ;
    \draw[dashed] (-3.6,-1) --(3.6,-1);
    \draw[dashed] (-3.6,1) --(3.6,1);
    \foreach \x in {-2,-1,0,1,2} {
       \draw (\x,0.1)--(\x,0) node[below]{$\x$};
    }
    \node  at (0,-1.5) {#2};
}


\newcommand{\CD}[2]{

    \draw[->] (-4,0)--(4,0) node[below] {$x$};
    \draw[->] (0,-1.3)--(0,1.5) node[right]{$y$};
    \draw[domain=-3.5:3.5,smooth,samples=200,thick] plot[id=#2] function { #1(x*pi)} ;
    \draw[dashed] (-3.6,-1) --(3.6,-1);
    \draw[dashed] (-3.6,1) --(3.6,1);
    \foreach \x in {-4,-2,0,2,4} {
       \draw (\x/2,0.1)--(\x/2,0) node[below]{$\x$};
    }
    \node  at (0,-1.5) {#2};
}
